Creacion de tablas

create TABLE POKEDEX
(

    ID_POKEMON NUMBER PRIMARY KEY NOT NULL,
    NOMBRE VARCHAR2(20) NOT NULL UNIQUE,
    TIPO VARCHAR2(25) NOT NULL,
    DESCRIPCION VARCHAR2(200) NOT NULL UNIQUE,
    RATIO NUMBER NOT NULL,
    VISTO varchar2(2char) NOT NULL,
    CAPTURA varchar2(2char) NOT NULL
);

CREATE TABLE CAPTURA
(
    ID_CAPTURA NUMBER PRIMARY KEY NOT NULL,
    ID_POKEBALL NUMBER REFERENCES POKEBALL(ID_POKEBALL) NOT NULL,
    ID_POKEMON NUMBER REFERENCES POKEDEX(ID_POKEMON) NOT NULL
);
/
ALTER TABLE  "CAPTURA" ADD FOREIGN KEY ("ID_POKEBALL")
	  REFERENCES  "POKEBALL" ("ID_POKEBALL") ENABLE
/
ALTER TABLE  "CAPTURA" ADD FOREIGN KEY ("ID_POKEMON")
	  REFERENCES  "POKEDEX" ("ID_POKEMON") ENABLE
/

CREATE TABLE RUTA
(

    ID_RUTA NUMBER PRIMARY KEY NOT NULL,
    NOMBRE VARCHAR2(20) NOT NULL UNIQUE,
    DESCRIPCION VARCHAR2(200) NOT NULL UNIQUE,
    VISITADA INTEGER NOT NULL
);


CREATE TABLE POKEBALL
(

    ID_POKEBALL NUMBER PRIMARY KEY NOT NULL,
    NOMBRE VARCHAR2(20) NOT NULL UNIQUE,
    DESCRIPCION VARCHAR2(200) NOT NULL UNIQUE,
    RATIO_CAPTURA NUMBER NOT NULL
);

CREATE TABLE  "APARECE" 
   (	"ID_APARECE" NUMBER NOT NULL ENABLE, 
	"ID_RUTA" NUMBER, 
	"ID_POKEMON" NUMBER, 
	 PRIMARY KEY ("ID_APARECE")
  USING INDEX  ENABLE
   )
/
ALTER TABLE  "APARECE" ADD FOREIGN KEY ("ID_RUTA")
	  REFERENCES  "POKEBALL" ("ID_POKEBALL") ENABLE
/
ALTER TABLE  "APARECE" ADD FOREIGN KEY ("ID_POKEMON")
	  REFERENCES  "POKEDEX" ("ID_POKEMON") ENABLE
/
